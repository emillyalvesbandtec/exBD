create database bdAtleta;
use bdAtleta;

create table tbAtleta(
id_atleta int primary key auto_increment,
nome_atleta varchar(40),
esporte varchar(40),
medalhas int
);

create table tbPais(
id_pais int primary key auto_increment,
nome_pais varchar(40),
capital varchar (40)
);
