create database bdMusica;
use bdMusica;
create table tbMusica(
id_Musica int primary key auto_increment,
nome_Musica varchar (40),
artista varchar (40),
genero varchar (40)
);

create table tbAlbum(
id_Album int primary key auto_increment,
nome_Album varchar (40),
gravadora varchar (40)
);

alter table tbMusica add column fk_album int;
